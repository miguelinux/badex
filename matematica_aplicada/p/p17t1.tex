\question Resuelve las siguientes ecuaciones y compruebalos sustituyendo el
          valor de $x$

  \begin{parts}
    \part $6x+4 = 3x+10$
      \begin{solution}[2cm]
          $6x-3x = 10 - 4$ \\
          $3x = 6$ \\
          $x = \frac{6}{3} = 2$
      \end{solution}
    \part $6x+3 = 2x+11$
      \begin{solution}[2cm]
          $6x-2x = 11-3$ \\
          $8x = 8$ \\
          $x = \frac{8}{8} = 1$
      \end{solution}
    \part $2x+5 = x-3$
      \begin{solution}[2cm]
          $2x-1x = -3-5$ \\
          $x = -8$
      \end{solution}
  \end{parts}
