\question Una campana que cuelga de un campanario tiene un peso de $1,200$
          \textbf{N}.¿Cúal es la fuerza que ejerce sobre el campanario?

  \begin{oneparchoices}
    \choice $0$ N
    \choice $1,200$ Kg
    \CorrectChoice $1,200$ N
    \choice $-1,200$ Kg
  \end{oneparchoices}
  \answerline[C]
